// 初始化数据库表
async function initDatabase() {
    const connection = await pool.getConnection();
    try {
        // 创建 drawings 表
        await connection.execute(`
            CREATE TABLE IF NOT EXISTS drawings (
                id INT AUTO_INCREMENT PRIMARY KEY,
                name VARCHAR(255) NOT NULL,
                type VARCHAR(50) NOT NULL COMMENT 'point/line/polygon/circle/rectangle',
                description TEXT,
                properties JSON,
                geometry JSON NOT NULL,
                center_lat DECIMAL(10, 8),
                center_lng DECIMAL(11, 8),
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
                updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
                INDEX idx_type (type),
                INDEX idx_location (center_lat, center_lng),
                SPATIAL INDEX idx_geometry (spatial_geometry)  -- MySQL 8.0+
            ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4
        `);
        
        // 添加空间几何列（MySQL 8.0+）
        try {
            await connection.execute(`
                ALTER TABLE drawings 
                ADD COLUMN IF NOT EXISTS spatial_geometry GEOMETRY
            `);
        } catch (e) {
            // 忽略已存在的错误
        }
        
        console.log('✅ 数据库初始化完成');
    } finally {
        connection.release();
    }
}

module.exports = { pool, initDatabase };

// ========== server/models/Drawing.js ==========
const { pool } = require('../config/database');

class Drawing {
    // 创建绘制记录
    static async create(data) {
        const { name, type, description, properties, geometry, center } = data;
        
        const [result] = await pool.execute(
            `INSERT INTO drawings 
             (name, type, description, properties, geometry, center_lat, center_lng, spatial_geometry) 
             VALUES (?, ?, ?, ?, ?, ?, ?, ST_GeomFromGeoJSON(?))`,
            [
                name,
                type,
                description || null,
                JSON.stringify(properties || {}),
                JSON.stringify(geometry),
                center?.lat || null,
                center?.lng || null,
                JSON.stringify(geometry)
            ]
        );
        
        return result.insertId;
    }
    
    // 批量保存
    static async createBatch(features) {
        const connection = await pool.getConnection();
        try {
            await connection.beginTransaction();
            
            const ids = [];
            for (const feature of features) {
                const { geometry, properties } = feature;
                const type = this.getGeometryType(geometry.type);
                const center = this.calculateCenter(geometry);
                
                const [result] = await connection.execute(
                    `INSERT INTO drawings 
                     (name, type, description, properties, geometry, center_lat, center_lng) 
                     VALUES (?, ?, ?, ?, ?, ?, ?)`,
                    [
                        properties?.name || `未命名${type}`,
                        type,
                        properties?.description || null,
                        JSON.stringify(properties || {}),
                        JSON.stringify(geometry),
                        center?.lat,
                        center?.lng
                    ]
                );
                ids.push(result.insertId);
            }
            
            await connection.commit();
            return ids;
        } catch (error) {
            await connection.rollback();
            throw error;
        } finally {
            connection.release();
        }
    }
    
    // 查询所有
    static async findAll(bounds = null) {
        let sql = 'SELECT * FROM drawings WHERE 1=1';
        const params = [];
        
        // 空间查询：范围内搜索
        if (bounds) {
            sql += ` AND center_lat BETWEEN ? AND ? AND center_lng BETWEEN ? AND ?`;
            params.push(bounds.south, bounds.north, bounds.west, bounds.east);
        }
        
        sql += ' ORDER BY created_at DESC';
        
        const [rows] = await pool.execute(sql, params);
        return rows.map(row => this.formatRow(row));
    }
    
    // 根据ID查询
    static async findById(id) {
        const [rows] = await pool.execute(
            'SELECT * FROM drawings WHERE id = ?',
            [id]
        );
        return rows[0] ? this.formatRow(rows[0]) : null;
    }
    
    // 更新
    static async update(id, data) {
        const { name, description, properties, geometry } = data;
        
        const [result] = await pool.execute(
            `UPDATE drawings 
             SET name = ?, description = ?, properties = ?, geometry = ?
             WHERE id = ?`,
            [
                name,
                description || null,
                JSON.stringify(properties || {}),
                JSON.stringify(geometry),
                id
            ]
        );
        
        return result.affectedRows > 0;
    }
    
    // 删除
    static async delete(id) {
        const [result] = await pool.execute(
            'DELETE FROM drawings WHERE id = ?',
            [id]
        );
        return result.affectedRows > 0;
    }
    
    // 搜索附近
    static async findNearby(lat, lng, radiusKm = 5) {
        // 使用 Haversine 公式计算距离
        const [rows] = await pool.execute(`
            SELECT *, 
                (6371 * acos(
                    cos(radians(?)) * cos(radians(center_lat)) * 
                    cos(radians(center_lng) - radians(?)) + 
                    sin(radians(?)) * sin(radians(center_lat))
                )) AS distance
            FROM drawings
            HAVING distance < ?
            ORDER BY distance
            LIMIT 50
        `, [lat, lng, lat, radiusKm]);
        
        return rows.map(row => ({
            ...this.formatRow(row),
            distance: parseFloat(row.distance.toFixed(2))
        }));
    }
    
    // 辅助方法：格式化返回数据
    static formatRow(row) {
        return {
            id: row.id,
            type: row.type,
            name: row.name,
            description: row.description,
            properties: typeof row.properties === 'string' 
                ? JSON.parse(row.properties) 
                : row.properties,
            geometry: typeof row.geometry === 'string' 
                ? JSON.parse(row.geometry) 
                : row.geometry,
            center: row.center_lat ? {
                lat: parseFloat(row.center_lat),
                lng: parseFloat(row.center_lng)
            } : null,
            createdAt: row.created_at,
            updatedAt: row.updated_at
        };
    }
    
    // 辅助方法：获取几何类型
    static getGeometryType(geojsonType) {
        const typeMap = {
            'Point': 'point',
            'LineString': 'line',
            'Polygon': 'polygon',
            'MultiPolygon': 'polygon',
            'Circle': 'circle'
        };
        return typeMap[geojsonType] || 'unknown';
    }
    
    // 辅助方法：计算中心点
    static calculateCenter(geometry) {
        const coords = geometry.coordinates;
        
        switch (geometry.type) {
            case 'Point':
                return { lat: coords[1], lng: coords[0] };
            case 'LineString':
                // 取中点
                const mid = Math.floor(coords.length / 2);
                return { lat: coords[mid][1], lng: coords[mid][0] };
            case 'Polygon':
                // 取第一个环的中心点近似
                const ring = coords[0];
                let lat = 0, lng = 0;
                ring.forEach(c => { lat += c[1]; lng += c[0]; });
                return { lat: lat/ring.length, lng: lng/ring.length };
            default:
                return null;
        }
    }
}

// ========== server/routes/drawings.js ==========
const express = require('express');
const router = express.Router();
const Drawing = require('../models/Drawing');

// GET /api/drawings - 获取所有绘制数据
router.get('/', async (req, res) => {
    try {
        const { south, north, west, east } = req.query;
        const bounds = south ? { south, north, west, east } : null;
        
        const drawings = await Drawing.findAll(bounds);
        res.json({
            success: true,
            count: drawings.length,
            data: drawings
        });
    } catch (error) {
        res.status(500).json({ success: false, error: error.message });
    }
});

// GET /api/drawings/nearby - 附近搜索
router.get('/nearby', async (req, res) => {
    try {
        const { lat, lng, radius = 5 } = req.query;
        
        if (!lat || !lng) {
            return res.status(400).json({ 
                success: false, 
                error: '缺少 lat 或 lng 参数' 
            });
        }
        
        const drawings = await Drawing.findNearby(
            parseFloat(lat), 
            parseFloat(lng), 
            parseFloat(radius)
        );
        
        res.json({
            success: true,
            count: drawings.length,
            data: drawings
        });
    } catch (error) {
        res.status(500).json({ success: false, error: error.message });
    }
});

// GET /api/drawings/:id - 获取单个
router.get('/:id', async (req, res) => {
    try {
        const drawing = await Drawing.findById(req.params.id);
        if (!drawing) {
            return res.status(404).json({ success: false, error: '未找到' });
        }
        res.json({ success: true, data: drawing });
    } catch (error) {
        res.status(500).json({ success: false, error: error.message });
    }
});

// POST /api/drawings - 保存（支持单条或批量）
router.post('/', async (req, res) => {
    try {
        const { features, type } = req.body;
        
        // 批量保存 GeoJSON FeatureCollection
        if (type === 'FeatureCollection' && Array.isArray(features)) {
            const ids = await Drawing.createBatch(features);
            return res.json({
                success: true,
                message: `成功保存 ${ids.length} 条数据`,
                ids: ids
            });
        }
        
        // 单条保存
        const id = await Drawing.create(req.body);
        res.json({
            success: true,
            message: '保存成功',
            id: id
        });
    } catch (error) {
        res.status(500).json({ success: false, error: error.message });
    }
});

// PUT /api/drawings/:id - 更新
router.put('/:id', async (req, res) => {
    try {
        const success = await Drawing.update(req.params.id, req.body);
        res.json({
            success: success,
            message: success ? '更新成功' : '未找到或无需更新'
        });
    } catch (error) {
        res.status(500).json({ success: false, error: error.message });
    }
});

// DELETE /api/drawings/:id - 删除
router.delete('/:id', async (req, res) => {

            message: success ? '删除成功' : '未找到'

            document.getElementById('coords').innerHTML = '纬度: ' + lat + ', 经度: ' + lng;
            
            L.circleMarker([lat, lng], {
                radius: 5,
                color: 'red'
            }).addTo(coordLayer)
              .bindPopup(lat + ', ' + lng)
              .openPopup();
        });
        
        // ===== 双击结束画线 =====
        map.on('dblclick', function(e) {
            // 取消单击事件
            if (clickTimer) {
                clearTimeout(clickTimer);
                clickTimer = null;
            }
            
            if (drawMode === 'line') {
                drawMode = '';
                console.log('✅ 画线完成，坐标:', JSON.stringify(linePoints));
                currentLine = null;
                alert('画线完成！共' + linePoints.length + '个点');
            }
        });
        
        // ===== 绘制处理函数 =====
        function handleDrawClick(e) {
            if (drawMode === 'point') {
                L.marker(e.latlng).addTo(drawLayer)
                    .bindPopup('自定义标记<br>纬度: ' + e.latlng.lat.toFixed(6) + '<br>经度: ' + e.latlng.lng.toFixed(6));
                drawMode = '';
                alert('标记已放置');
            }
            else if (drawMode === 'line') {
                linePoints.push(e.latlng);
                
                // 删除旧线，画新线（只有一条）
                if (currentLine) {
                    drawLayer.removeLayer(currentLine);
                }
                
                currentLine = L.polyline(linePoints, {
                    color: 'blue',
                    weight: 4
                }).addTo(drawLayer);
                
                // 显示点数
                document.getElementById('coords').innerHTML = 
                    '画线中: 已点击 ' + linePoints.length + ' 个点（双击结束）';
            }
        }
        
        // ===== 按钮函数 =====
        function startDrawPoint() {
            drawMode = 'point';
            document.getElementById('coords').innerHTML = '点击地图放置标记';
        }
        
        function startDrawLine() {
            drawMode = 'line';
            linePoints = [];
            currentLine = null;
            document.getElementById('coords').innerHTML = '点击地图开始画线（双击结束）';
        }
        
        function clearDraw() {
            drawLayer.clearLayers();
            coordLayer.clearLayers();
            linePoints = [];
            currentLine = null;
            drawMode = '';
            document.getElementById('coords').innerHTML = '纬度: --, 经度: --';
        }
    </script>
